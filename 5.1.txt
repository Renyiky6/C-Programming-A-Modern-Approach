//输入整数（四位及以内）  并判断位数


#include<stdio.h>

int main(void) 
{
	int x;
	printf("请输入一个数字（五位以下）: ");
	scanf_s("%d", &x);
	printf("位数是：");
	if (x >= -9 && x <= 9)
		printf("1");
	else if ((x <= -10 && x >= -99) || (x >= 10 && x <= 99))
		printf("2");
	else if ((x <= -100 && x >= -999) || (x >= 100 && x <= 999))
		printf("3");
	else
		printf("4");
	return 0;
}
