//输入风速 显示风力等级


#include<stdio.h>

int main(void) 
{
	float x;
	printf("please enter the speed : ");
	scanf_s("%f", &x);
	printf("The scale of the wind is :");
	if (x < 1.0f) {
		printf("Calm");
	}
	else if (x >= 1.0f && x < 4.0f)
		printf("Light air\n");
	else if (x >= 4.0f && x < 28.0f)
		printf("Breeze\n");
	else if (x >= 28.0f && x < 48.0f)
		printf("Gale\n");
	else if (x >= 48.0f && x < 63.0f)
		printf("Storm\n");
	else
		printf("Hurricane\n");

	return 0;
}
