//输入收入 输出应缴税额


#include<stdio.h>

int main(void) 
{
	float x;
	float tax;
	printf("Please enter yout income($):");
	scanf_s("%f", &x);
	if (x < 750.0f)
		tax = x*0.01;
	else if (x >= 750.0f&&x <= 2250.0f)
		tax = 7.50 + (x - 750.0f)*0.02;
	else if (x >= 2250.0f&&x <= 3750.0f)
		tax = 37.50 + (x - 2250.0f)*0.03;
	else if (x >= 3750.0f&x <= 5250.0f)
		tax = 82.50 + (x - 3750.0f)*0.04;
	else if (x >= 5250.0f&&x <= 7000.0f)
		tax = 142.5 + (x - 5250.0f)*0.05;
	else
		tax = 230.0f + (x - 7000.0f)*0.06;

	printf("Your tax is: $%.2f\n", tax);
	return 0;
}
