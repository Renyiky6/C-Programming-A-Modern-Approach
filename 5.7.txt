//输入分数 判断等级


#include<stdio.h>

int main(void) 
{
	int a, b, c, d;
	int max,min;

	printf("Enter four integers:");
	scanf_s("%d%d%d%d", &a, &b, &c, &d);
	max = a;
	if (max < b)
		max = b;
	if (max < c)
		max = c;
	max = max < d ? d : max;

	min = a;
	if (min > b)
		min = b;
	if (min > c)
		min = c;
	min = min > d ? d : min;

	printf("Largest: %d\nSmallest: %d\n", max, min);


	return 0;
}
